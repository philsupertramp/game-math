538  for(const auto& c : split(strip(processString))) {
539  if(isNumber(c)) {

544  } else if(isSymbol(c)) {

546  symbol->isNegative = nextIsNegative;
547  nextIsNegative = false;
548  operandStack.push_back(symbol);
549  if(!HasSymbol(symbol)) symbols.push_back(symbol);
550  } else if(isParenthesesOpen(c)) {
551  operatorStack.push_back(c);
552  } else if(isOperator(c)) {

554  if(prevWasOperator && currentOp->value[0] == '-') {
555  nextIsNegative = true;

576  } else {
577  break;

579  }
580  }
581  operatorStack.push_back(c);
582  }

587

589  operandStack.pop_back();
590

592  operandStack.pop_back();

599  } else

605  } else {

608  return nullptr;

611  }
