161 assert(columns % 2 == 0);
162
163 constexpr double two_pi = 2.0 * M_PI;
164 Matrix out;
165 out.Resize(rows, columns);

167 for(size_t k = 0; k < columns; k += 2) {
168 auto u1 = Random::Get();
169 auto u2 = Random::Get();
170
171 auto mag = sigma * sqrt(-2.0 * log(u1));
172 out(i, k) = mag * cos(two_pi * u2) + mu;

174 }
