191 void AddData(
192 const Matrix<double>& mat, const std::string& name, DataTypes dataTypeName = DataTypes::NONE, int dimensions = 2) {
193 if(dimensions <= 3) {
194 // calculate boundaries
195 auto new_bX = getBoundaries(mat.GetSlice(0, mat.rows() - 1, 0, 0));
196 auto new_bY = getBoundaries(mat.GetSlice(0, mat.rows() - 1, 1, 1));
197
198 bX.min = bX.min > new_bX.min ? new_bX.min : bX.min;
199 bX.max = bX.max < new_bX.max ? new_bX.max : bX.max;
200
201 bY.min = bY.min > new_bY.min ? new_bY.min : bY.min;
202 bY.max = bY.max < new_bY.max ? new_bY.max : bY.max;
203 } else {
204 bX.min = 0;
205 bY.min = 0;
206 bX.max = mat.columns();
207 bY.max = mat.columns();
208 }
209
210 // append storage file
211 storageFile = fopen(dataFileName, "a");
212 if(storageFile == nullptr) {
213 // error
214 return;
215 }
216 fprintf(storageFile, "# ");
217 fprintf(storageFile, "%s", dataFileName);
218 fprintf(storageFile, "\n");
219
220 for(size_t i = 0; i < mat.columns(); i += dimensions) {
221 fprintf(storageFile, "# X Y Z\n");
222 for(size_t j = 0; j < mat.rows(); ++j) {
223 for(int k = 0; k < dimensions; ++k) { fprintf(storageFile, "%g ", mat(j, i + k)); }
224 fprintf(storageFile, "\n");
225 }
226 fprintf(storageFile, "\n\n");
227 }
228 fclose(storageFile);
229
230 char* newName;
231 newName = new char[name.size() + 1];
232 std::copy(name.begin(), name.end(), newName);
233 newName[name.size()] = '\0';
234 PlotIndex index = { 0, -1 };
235 if(!attributes.plotIndices.empty()) {
236 index = attributes.plotIndices[attributes.plotIndices.size() - 1];
237 index.start++;
238 }
239
240 attributes.plotIndices.push_back(index);
241 attributes.plotNames.push_back(newName);
