85 Matrix<double> curv(double h) {

87 auto dim = XI.rows() - 2;
88 auto rhs = 6.0 / (h * h) * (tridiag(dim, dim, 1, -2, 1) * YI.GetSlice(1, YI.rows() - 2, 0, YI.columns() - 1));
