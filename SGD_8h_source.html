99 double update_weights(const Matrix<double>& xi, const Matrix<double>& target, Matrix<double>& weights) const {
100 auto output = Matrix<double>();
101 if(net_input_fun != nullptr) {
102 output = net_input_fun(xi);
103 } else {
104 output = netInput(xi, weights);
105 }
106 auto error = target - output;
107 weights.SetRow(0, weights(0) + eta * error);
108 auto delta = eta * (xi.Transpose() * error);
109 for(size_t i = 0; i < weights.rows() - 1; i++) { weights.SetRow(i + 1, weights(i + 1) + delta(i)); }
110 return ((error * error) * 0.5).sumElements() / (double)target.rows();

130 [[nodiscard]] static Matrix<double> netInput(const Matrix<double>& X, const Matrix<double>& weights) {
131 Matrix<double> A(0, weights.rows() - 1, weights.columns());
132 Matrix<double> B(0, 1, weights.columns());
133 for(size_t i = 0; i < weights.rows(); i++) {
134 for(size_t j = 0; j < weights.columns(); j++) {
135 if(i == 0) {
136 B(i, j) = weights(i, j);
137 } else {
138 A(i - 1, j) = weights(i, j);
139 }
140 }
141 }

143 // Note: since Matrix<T> will never allow Matrix-Scalar Addition
144 // we need to create a vector of size of rows of the input
145 // values (jeez, horrible sentence). Therefore we rescale
146 // B and assign the bias to each element.
147 B = Matrix<double>(B(0, 0), X.rows(), 1);
148 return (X * A) + B;
