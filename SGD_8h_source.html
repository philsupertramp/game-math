95  double update_weights(const Matrix<double>& xi, const Matrix<double>& target, Matrix<double>& weights) const {
96  auto output = Matrix<double>();
97  if(net_input_fun != nullptr) {
98  output = net_input_fun(xi);
99  } else {
100  output = netInput(xi, weights);
101  }
102  auto error = target - output;
103  weights.SetRow(0, weights(0) + eta * error);
104  auto delta = eta * (xi.Transpose() * error);
105  for(size_t i = 0; i < weights.rows() - 1; i++) { weights.SetRow(i + 1, weights(i + 1) + delta(i)); }
106  return ((error * error) * 0.5).sumElements() / (double)target.rows();
107  }

126  [[nodiscard]] static Matrix<double> netInput(const Matrix<double>& X, const Matrix<double>& weights) {
127  Matrix<double> A(0, weights.rows() - 1, weights.columns());
128  Matrix<double> B(0, 1, weights.columns());
129  for(size_t i = 0; i < weights.rows(); i++) {
130  for(size_t j = 0; j < weights.columns(); j++) {
131  if(i == 0) {
132  B(i, j) = weights(i, j);
133  } else {
