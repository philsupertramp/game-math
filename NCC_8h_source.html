81 void fit_batch(const Matrix<double>& X, const Matrix<double>& y, const Matrix<double>& labels) {
82 // TODO: Technically we can omit labels here and only pass the number of unique labels. Or we keep a vector
83 // of unique labels as a class attribute. Both options might make sense.
84 // O(K)
85 for(size_t i = 0; i < labels.rows(); ++i) {
86 std::function<bool(double)> condition = [i](double x) { return bool(x == i); };
87 // O(N)
88 auto yis = where(condition, y, { { 1 } }, { { 0 } });
89 // O(N)
90 auto Nk = yis.sumElements();
91 // O(2N)
92 auto Xis = X.GetSlicesByIndex(where_true(yis));
93 // O(N * N * N)
94 weights.SetSlice(i, Xis.sum(1) * (1. / Nk));
95 }
96 }

110 void fit_iterative(const Matrix<double>& X, const Matrix<double>& y, const Matrix<double>& labels) {
111 auto counters = Matrix<int>(0, labels.rows(), 1);
112 // O(N)
113 for(size_t i = 0; i < X.rows(); ++i) {
114 auto k = y(i, 0);
115 auto xi = X.GetSlice(i, i);
116 auto current_counter = counters(k, 0);
117 // O(D)
118 weights.SetSlice(
119 k,
120 // O(D)
121 weights.GetSlice(k, k) * (current_counter / double(current_counter + 1))
122 // O(D)
123 + xi * (1. / double(current_counter + 1)));
124 counters(k, 0) += 1;
125 }
126 }
